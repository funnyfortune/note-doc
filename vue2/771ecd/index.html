API 请求

介绍

其中，@/src/utils/request.js 是基于 axios 的封装，便于统一处理 POST，GET 等请求参数，请求头，以及错误提示信息等。 它封装了全局request拦截器、response拦截器、统一的错误处理、统一做了超时处理、baseURL设置等。

Axios 是什么?

Axios 是一个基于 promise (opens new window) 网络请求库，作用于node.js (opens new window) 和浏览器中。 它是 isomorphic (opens new window) 的(即同一套代码可以运行在浏览器和node.js中)。在服务端它使用原生 node.js http 模块, 而在客户端 (浏览端) 则使用 XMLHttpRequests。

特性

  • 从浏览器创建 XMLHttpRequests (opens new window)

  • 从 node.js 创建 http (opens new window) 请求

  • 支持 Promise (opens new window) API

  • 拦截请求和响应

  • 转换请求和响应数据

  • 取消请求

  • 自动转换JSON数据

  • 客户端支持防御XSRF (opens new window)

来自 Axios官网 (opens new window)

# 交互流程

一个完整的前端 UI 交互到服务端处理流程是这样的：

每个 API 方法，会调用 request 方法，发起对后端 RESTful API 的调用。

为了方便管理维护，统一的请求处理都放在 @/src/api文件夹中，并且一般按照 @views/模块 维度进行拆分文件，如：

# 请求示例

import request from '@/utils/request'

// 查询菜单详细
export function getMenu(menuId) {
 return request({
   url: '/sys/menu/' + menuId,
   method: 'get'
 })
}

// 新增菜单
export function addMenu(data) {
 return request({
   url: '/sys/menu',
   method: 'post',
   data: data
 })
}

// 修改菜单
export function updateMenu(data) {
 return request({
   url: '/sys/menu',
   method: 'put',
   data: data
 })
}

// 删除菜单
export function delMenu(menuId) {
 return request({
   url: '/sys/menu/' + menuId,
   method: 'delete'
 })
}

# request.js 解析

@/utils/request基于 axios (opens new window) 封装，统一对http请求,对请求参数、请求头，以及响应拦截过滤，错误提示信息等进行处理。

注意

以下是源码的解析，业务开发中，不需要对request.js进行二次开发。

# 创建 axios 实例

import axios from 'axios'

// 创建axios实例
const service = axios.create({
  // axios中请求配置有baseURL选项，表示请求URL公共部分
  baseURL: process.env.VUE_APP_BASE_API,
  timeout: 1000 * 30,
  withCredentials: true
})

  • baseURL 将自动加在 url 前面，除非 url 是一个绝对 URL。
  • timeout 如果请求时间超过 timeout 的值，则请求会被中断,默认值是 0 (永不超时)。
  • withCredentials 表示跨域请求时是否需要使用凭证。

更多配置信息，请查看官方文档 (opens new window)配置信息。

# Request 拦截器

  • Authorization 、 tenant-id 请求头

  • GET 请求参数的拼接

// request拦截器
service.interceptors.request.use(
  config => {
    if(config.data){
      config.headers['Content-Type'] = 'application/json'
    }
    if (getToken()) {
      config.headers['Authorization'] = TOKEN_PREFIX + getToken() // 让每个请求携带自定义token 请根据实际情况自行修改
    }
    return config
  },
  error => {
    return Promise.reject(error)
  }
)

# Response 拦截器

  • Token 失效、登录过期时，跳回首页

  • 请求失败，Message 错误提示

// 响应拦截器
service.interceptors.response.use(res => {
   console.log(res.request.responseURL,res)
 if(res.headers['content-type'].indexOf('json') == -1){
   return res
 }
   const code = res.data.code
   if (code === 401) {
     MessageBox.confirm(
       '登录状态已过期，您可以继续留在该页面，或者重新登录',
       '系统提示',
       {
         confirmButtonText: '重新登录',
         cancelButtonText: '取消',
         type: 'warning'
       }
     ).then(() => {
       store.dispatch('LogOut').then(() => {
         location.reload() // 为了重新实例化vue-router对象 避免bug
       })
     })
   } else if (code !== 200) {
     let msg = '系统繁忙'
     if(res.data.msg && res.data.msg !=null && res.data.msg != ''){
       msg = res.data.msg
     }
     Message({
       message: code >= 400 && code<600? '服务异常':msg,
       center: true,
       type: 'error'
     });
     return Promise.reject('error')
   } else {
     return res.data
   }
 },
 error => {
   Message({
     message: '网络异常，稍后再试',
     type: 'error',
     duration: 5 * 1000
   })
   return Promise.reject(error)
 }
)

# 文件下载

/**
 *
 * @param url 下载地址
 * @param query 请求参数
 * @param fileName 文件名
 *
 */
service.downloadFile = ( url,query,fileName) => {
  service({
    url: url,
    method: 'get',
    responseType:'blob',
    params: query
  }).then(res=>{
    let blob = new Blob([res.data]);
    let downloadElement = document.createElement("a");
    let href = window.URL.createObjectURL(blob);
    downloadElement.href = href;
    downloadElement.download = fileName;
    document.body.appendChild(downloadElement);
    downloadElement.click();
    document.body.removeChild(downloadElement);
    window.URL.revokeObjectURL(href);
  })
}
